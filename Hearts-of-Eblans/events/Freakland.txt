			factor = 50

				factor = 300
        		tag = MNY
			}
			modifier = {
				factor = 70
        		tag = BTK
			}
		}

		hidden_effect = {
			if = {
				limit = {
					tag = MNY
				}
				set_variable = { THIS.new_freakland_level = 2 }
				set_freakland_level = yes
			}
			if = {
				limit = {
					tag = BTK
				}
				set_variable = { THIS.new_freakland_level = 3 }
				set_freakland_level = yes
			}
		}

		THIS = {
            add_to_array = {
                array = global.freakland_members
                value = THIS.id
            }
			custom_effect_tooltip = new_freakmember_tt

			custom_effect_tooltip = generic_skip_one_line_tt

			custom_effect_tooltip = {
				localization_key = squad_decision_willopen_tt
				DECISION_NAME = freakland_decisions_category
			}

			hidden_effect = {
				update_freakland_idea = yes
				news_event = { id = freakland.4 days = 5 }
			}
        }
	}

	option = {
		name = freakland.5.b
		ai_chance = {
			factor = 20
		}
